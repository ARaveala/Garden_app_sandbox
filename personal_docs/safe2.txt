import { chromium } from 'playwright';
import { BaseDataHarvester } from './base.js';

export class USDAScra
per extends BaseDataHarvester {
  async getPlantList(limit = 10) {
    const browser = await chromium.launch({...});
    const page = await browser.newPage();
    
    const url = `${this.baseUrl}${this.config.endpoints.plant_list}`;
    await page.goto(url, { waitUntil: 'networkidle' });
    
    const selector = this.config.selectors.plant_links;
    const plantSymbols = await page.evaluate((selector, limit) => {
      const links = Array.from(document.querySelectorAll(selector));
      return links.slice(0, limit)
        .map(link => {
          const match = link.href.match(/plant-profile\/([A-Z0-9]+)/);
          return match ? match[1] : null;
        })
        .filter(Boolean);
    }, selector, limit);
    
    await browser.close();
    return plantSymbols;
  }
  
  async harvestPlant(symbol, dataType = 'traits') {
    const browser = await chromium.launch({...});
    const page = await browser.newPage();
    
    // Use config endpoint
    const endpoint = this.config.endpoints.plant_detail.replace('{symbol}', symbol);
    const url = `${this.baseUrl}${endpoint}`;
    
    await page.goto(url, { waitUntil: 'networkidle' });
    
    const selector = this.config.selectors.trait_rows;
    const traits = await page.evaluate((selector) => {
      const rows = Array.from(document.querySelectorAll(selector));
      const data = {};
      for (const row of rows) {
        const key = row.querySelector('th')?.innerText;
        const val = row.querySelector('td')?.innerText;
        if (key && val) data[key] = val;
      }
      return data;
    }, selector);
    
    await browser.close();
    
    return {
      source: 'usda',
      plant_identifier: symbol,
      data_type: this.normalizeDataType(dataType),
      raw_json: traits
    };
  }
}

import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';
import { USDADataHarvester } from './DataHarvesters/usda.js';
import { insertHarvest } from './database/connection.js';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

async function main() {
  // Load config
  const configPath = path.join(__dirname, '../config/sources.json');
  const config = JSON.parse(fs.readFileSync(configPath, 'utf8'));
  
  // Initialize DataHarvester
  const DataHarvester = new USDADataHarvester(config.usda);
  
  // Get plants to harvest
  console.log('📋 Fetching plant list...');
  const plantSymbols = await DataHarvester.getPlantList(3);
  console.log(`✓ Found ${plantSymbols.length} plants:`, plantSymbols);
  
  // Harvest all concurrently
  console.log('🌱 Scraping plants...');
  const results = await Promise.all(
    plantSymbols.map(symbol => DataHarvester.harvestPlant(symbol))
  );
  
  // Insert into database
  console.log('💾 Saving to database...');
  for (const result of results) {
    const inserted = await insertHarvest(result);
    if (inserted) {
      console.log(`✓ Saved ${result.plant_identifier}`);
    } else {
      console.log(`⊘ ${result.plant_identifier} already exists`);
    }
  }
  
  console.log('✅ Done!');
  process.exit(0);
}

main().catch(err => {
  console.error('❌ Error:', err);
  process.exit(1);
});

//db connection 
import mysql from 'mysql2/promise';

let connection;

export async function getConnection() {
  if (!connection) {
    connection = await mysql.createConnection({
      host: process.env.DB_HOST,
      user: process.env.DB_USER,
      password: process.env.DB_PASS,
      database: process.env.DB_NAME
    });
  }
  return connection;
}

export async function insertHarvest(data) {
  const conn = await getConnection();
  
  try {
    await conn.execute(
      'INSERT INTO raw_harvest (source, plant_identifier, data_type, raw_json) VALUES (?, ?, ?, ?)',
      [data.source, data.plant_identifier, data.data_type, JSON.stringify(data.raw_json)]
    );
    return true; // Inserted
  } catch (err) {
    if (err.code === 'ER_DUP_ENTRY') {
      return false; // Already exists
    }
    throw err; // Real error
  }
}

export async function closeConnection() {
  if (connection) {
    await connection.end();
    connection = null;
  }
}
