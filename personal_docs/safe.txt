// Get plant symbols from search results
async function getPlantList(limit = 3) {
  const browser = await chromium.launch({...});
  const page = await browser.newPage();
  
  // USDA browse page (or search page)
  await page.goto('https://plants.usda.gov/home/plantProfile?sort=sciName', {
    waitUntil: 'networkidle'
  });
  
  // Extract plant symbols from the table/list
  const plantSymbols = await page.evaluate((limit) => {
    const links = Array.from(document.querySelectorAll('a[href*="/plant-profile/"]'));
    return links
      .slice(0, limit)
      .map(link => {
        const match = link.href.match(/plant-profile\/([A-Z0-9]+)/);
        return match ? match[1] : null;
      })
      .filter(Boolean);
  }, limit);
  
  await browser.close();
  return plantSymbols; // ['ABBA', 'ACPL', 'ACRU']
}

async function main() {
  const connection = await mysql.createConnection({...});
  
  // Get list of plants to scrape
  console.log('Fetching plant list...');
  const plantSymbols = await getPlantList(3); // Get 3 plants
  console.log('Found plants:', plantSymbols);
  
  // Scrape all plants CONCURRENTLY
  console.log('Scraping plants...');
  const results = await Promise.all(
    plantSymbols.map(symbol => scrapePlant(symbol))
  );
  
  // Insert into DB
  for (const result of results) {
    try {
      await connection.execute(
        'INSERT INTO raw_harvest (source, plant_identifier, data_type, raw_json) VALUES (?, ?, ?, ?)',
        ['usda', result.symbol, 'traits', JSON.stringify(result.traits)]
      );
      console.log(`✓ Inserted ${result.symbol}`);
    } catch (err) {
      if (err.code === 'ER_DUP_ENTRY') {
        console.log(`⊘ ${result.symbol} already exists, skipping`);
      } else {
        throw err;
      }
    }
  }
  
  await connection.end();
  console.log('Done!');
  process.exit(0);
}
